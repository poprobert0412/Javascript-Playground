// === TO-DO LIST ===
// Selectăm elementele necesare
const todoInput = document.getElementById("todoInput");
const todoList = document.getElementById("todoList");
const todoAdd = document.getElementById("todoAdd");

// Array care ține toate sarcinile
let todos = [];

// Funcție: adaugă o sarcină nouă
function addTodo() {
    const text = todoInput.value.trim();
    if (!text) return;

    todos.push({ text, done: false });
    todoInput.value = "";
    renderTodos();
}

// Funcție: afișează toate sarcinile
function renderTodos() {
    todoList.innerHTML = "";
    todos.forEach((todo, i) => {
        const div = document.createElement("div");
        div.className = `todo-item ${todo.done ? "done" : ""}`;
        div.innerHTML = `
            <span class="todo-check" onclick="toggleTodo(${i})">✓</span>
            <span class="todo-text">${todo.text}</span>
            <button class="todo-delete" onclick="deleteTodo(${i})">✕</button>
        `;
        todoList.appendChild(div);
    });
    todoCount.textContent = `${todos.length} sarcini (${todos.filter(t => t.done).length} complete)`;
}

// Toggle done, delete, clear
function toggleTodo(i) { todos[i].done = !todos[i].done; renderTodos(); }
function deleteTodo(i) { todos.splice(i, 1); renderTodos(); }

todoAdd.addEventListener("click", addTodo);
todoInput.addEventListener("keydown", (e) => { if (e.key === "Enter") addTodo(); });

// === QUIZ GAME ===
const questions = [
    { q: "Câte continente sunt?", a: ["5", "6", "7", "8"], c: 2 },
    { q: "Ce limbaj rulează în browser?", a: ["Python", "JavaScript", "C++", "Java"], c: 1 },
    { q: "Ce returnează typeof []?", a: ["array", "object", "list", "undefined"], c: 1 },
    // ... mai multe întrebări
];

let currentQ = 0, score = 0;

function showQuestion() {
    const q = questions[currentQ];
    document.getElementById("miniQuizQ").textContent = q.q;
    q.a.forEach((ans, i) => {
        const btn = document.getElementById(["mqA","mqB","mqC","mqD"][i]);
        btn.textContent = ans;
        btn.className = "quiz-answer-btn";
    });
}

function miniQuizAnswer(idx) {
    const correct = questions[currentQ].c;
    // Marchează corect/greșit cu classList
    if (idx === correct) { score++; /* .correct-ans */ }
    // Treci la următoarea întrebare
    currentQ++;
    if (currentQ < questions.length) showQuestion();
    else /* afișează scorul final */;
}

// === WEATHER APP cu Fetch API ===
// Folosim wttr.in — API gratis, fără API key!

async function getWeather(city) {
    try {
        // Fetch — face un request HTTP GET
        const response = await fetch(
            `https://wttr.in/${city}?format=j1`
        );

        // Verifică dacă request-ul a reușit
        if (!response.ok) {
            throw new Error("Orașul nu a fost găsit!");
        }

        // Parsează JSON din response
        const data = await response.json();

        // Extrage datele din JSON (destructuring!)
        const { temp_C, humidity, weatherDesc } =
            data.current_condition[0];

        // Afișează rezultatul
        document.getElementById("weatherTemp")
            .textContent = `${temp_C}°C`;
        document.getElementById("weatherCity")
            .textContent = city;
        document.getElementById("weatherDetails")
            .textContent = `${weatherDesc[0].value} • Umiditate: ${humidity}%`;

    } catch (err) {
        alert(`Eroare: ${err.message}`);
    }
}

// Event listener pe buton
weatherBtn.addEventListener("click", () => {
    const city = weatherInput.value.trim();
    if (city) getWeather(city);
});
