📦 Variabile

const x = 5;      // NU se reasignează
let y = 10;        // se poate reasigna
var z = 15;        // ❌ vechi, nu folosi

y = 20;            // ✅ OK
// x = 10;         // ❌ ERROR!
💡 Folosește const mereu. let doar când valoarea se schimbă.

🏷️ Tipuri de Date

"text"             // string
42                 // number
true / false       // boolean
null               // nimic (intenționat)
undefined          // nedefinit
[1, 2, 3]          // array (object)
{a: 1}             // object

typeof "text"      // "string"
typeof null        // "object" (bug!)

🔀 Condiții

if (x > 0) {
    // pozitiv
} else if (x < 0) {
    // negativ
} else {
    // zero
}

// Ternary (pe o linie)
const r = x > 0 ? "da" : "nu";

// Switch
switch(zi) {
    case "Luni": /*...*/ break;
    default: /*...*/
}

🔄 Bucle

// FOR — știi câte iterații
for (let i = 0; i < 5; i++) {}

// WHILE — condiție
while (x > 0) { x--; }

// FOR...OF — array
for (const item of array) {}

// FOR...IN — chei obiect
for (const key in obj) {}

// break = STOP | continue = SKIP

⚙️ Funcții

// Clasică
function suma(a, b) {
    return a + b;
}

// Arrow (scurtă)
const suma = (a, b) => a + b;

// Arrow (cu body)
const salut = (n) => {
    return `Bună, ${n}!`;
};

// Default params
function f(x = 10) {}

📋 Array Methods

arr.push(x)        // adaugă la sfârșit
arr.pop()           // scoate ultimul
arr.unshift(x)      // adaugă la început
arr.length          // câte elemente

arr.map(n => n * 2)    // transformă
arr.filter(n => n > 5) // filtrează
arr.reduce((s,n) => s+n, 0) // reduce
arr.forEach(n => {})   // parcurge
arr.find(n => n > 5)   // primul match
arr.includes(5)        // conține?

📝 String Methods

str.length           // lungime
str.toUpperCase()    // MAJUSCULE
str.toLowerCase()    // minuscule
str.includes("x")   // conține?
str.startsWith("x") // începe cu?
str.indexOf("x")    // poziția
str.slice(0, 5)     // taie
str.replace("a","b")// înlocuiește
str.split(",")      // → array
str.trim()          // taie spații
`${var}`             // template literal

🧱 Obiecte

const obj = {
    name: "Robert",
    age: 25,
    greet() {
        return `Salut, ${this.name}!`;
    }
};

obj.name            // "Robert"
obj["age"]          // 25
Object.keys(obj)    // ["name","age"]
Object.values(obj)  // ["Robert", 25]
Object.entries(obj) // perechi

⚖️ Comparații & Logică

===  strict egal (tip + valoare)
!==  strict diferit
==   egal (convertește tipuri) ❌
>  <  >=  <=

&&   AND (ambele true)
||   OR (cel puțin unul true)
!    NOT (neagă)

x ?? "default"   // nullish coal.
obj?.prop         // optional chain
💡 Folosește MEREU === în loc de ==

🛡️ Error Handling

try {
    // cod riscant
    JSON.parse("invalid");
} catch (err) {
    console.log(err.message);
} finally {
    // se execută MEREU
}

// Aruncă eroare custom
throw new Error("Mesaj!");

🧰 Built-in Objects

// Math
Math.random()      // 0-1
Math.floor(4.7)    // 4
Math.ceil(4.2)     // 5
Math.max(3, 7)     // 7

// Date
new Date()
date.getFullYear()

// JSON
JSON.stringify(obj) // obj → string
JSON.parse(str)     // string → obj

// localStorage
localStorage.setItem("k", "v")
localStorage.getItem("k")

🔒 Scope & Closures

// Global — vizibil oriunde
const g = "global";

function f() {
    // Local — doar în funcție
    const l = "local";
}

// Block scope (let/const)
if (true) {
    let b = "block only";
}

// Closure — funcție cu "memorie"
function contor() {
    let n = 0;
    return () => ++n;
}

📦 Destructuring & Spread

// Obiect
const {name, age} = person;
const {name: n} = person; // redenumire

// Array
const [a, b, c] = [1, 2, 3];
const [first, ...rest] = arr;

// Spread
const arr2 = [...arr, 4, 5];
const obj2 = {...obj, new: "prop"};

// Rest params
function f(...args) {}

🏛️ Classes

class Animal {
    constructor(name) {
        this.name = name;
    }
    speak() {
        return `${this.name} vorbește`;
    }
}

class Dog extends Animal {
    constructor(name, breed) {
        super(name);
        this.breed = breed;
    }
}
const r = new Dog("Rex", "Husky");

⏳ Async/Await

// Promise
const p = new Promise((res, rej) => {
    res("done");
});
p.then(r => console.log(r));

// Async/Await
async function getData() {
    try {
        const r = await fetch(url);
        const data = await r.json();
        return data;
    } catch (err) {
        console.log(err);
    }
}

// Parallel
await Promise.all([f1(), f2()]);

🌐 DOM

document.getElementById("id")

document.querySelectorAll("div")

parent.appendChild(div)

🎯 Events

// Adaugă listener
el.addEventListener("click", (e) => {

// Evenimente comune

🌍 Fetch API

// GET
const r = await fetch(url);
const data = await r.json();

// POST
await fetch(url, {
    method: "POST",
    headers: {
        "Content-Type": "application/json"
    },
    body: JSON.stringify({key: "val"})
});

// Verificare
if (!response.ok) {
